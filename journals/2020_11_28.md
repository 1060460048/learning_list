## 搞懂 logseq 的基本使用，入门门槛略微有些高，提示不足
下面一道编程题：一块钱买一瓶饮料，三个瓶盖换一瓶饮料
```js
function buy(money) {
  function buy(money, caps, bottles) {
    return money === 0 && caps < 3 ? 
      bottles : 
      caps < 3 ? 
      buy(0, caps + money, bottles + money) : 
      buy(money + caps / 3 | 0, caps % 3, bottles)
  }
  return buy(money, 0, 0)
}
```
